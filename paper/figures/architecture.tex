\documentclass[tikz, border=6pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{positioning, arrows.meta, fit, backgrounds, calc, shapes.geometric}

% Color palette
\definecolor{encblue}{RGB}{52, 101, 164}
\definecolor{fpngreen}{RGB}{78, 153, 75}
\definecolor{seorange}{RGB}{206, 92, 0}
\definecolor{headwt}{RGB}{39, 116, 39}
\definecolor{headtc}{RGB}{143, 89, 2}
\definecolor{headet}{RGB}{92, 53, 102}
\definecolor{lightgray}{RGB}{238, 238, 236}
\definecolor{darkgray}{RGB}{85, 87, 83}

\tikzset{
  % Generic block
  blk/.style 2 args={
    rectangle, rounded corners=3pt, text centered,
    minimum width=#1, minimum height=0.58cm,
    font=\footnotesize\bfseries, inner sep=3pt,
    fill=#2!15, draw=#2, thick,
  },
  encblk/.style={blk={2.1cm}{encblue}},
  fpnblk/.style={blk={1.9cm}{fpngreen}},
  seblk/.style={blk={2.0cm}{seorange}},
  wtblk/.style={blk={2.2cm}{headwt}},
  tcblk/.style={blk={2.2cm}{headtc}},
  etblk/.style={blk={2.2cm}{headet}},
  outblk/.style={blk={1.3cm}{darkgray}},
  iobox/.style={rectangle, rounded corners=3pt, minimum width=1.7cm, minimum height=0.58cm,
                text centered, font=\footnotesize, fill=lightgray, draw=darkgray, thick},
  % Arrows
  arr/.style={-{Stealth[length=4.5pt]}, thick, color=darkgray!70},
  cond/.style={-{Stealth[length=4.5pt]}, dashed, thick, color=seorange!90},
  fpnarr/.style={-{Stealth[length=4.5pt]}, thick, color=fpngreen!70},
  % Group boxes
  grpbox/.style={rectangle, rounded corners=5pt, draw=#1!60, thick, fill=#1!4, inner sep=5pt},
  lbl/.style={font=\tiny, color=darkgray!80},
}

\begin{document}
\begin{tikzpicture}[node distance=0.38cm and 0.7cm]

% ── INPUT ────────────────────────────────────────────────────────────────
\node[iobox] (input) {Input\\[-3pt]\tiny $4{\times}H{\times}W{\times}D$};

% ── ENCODER ──────────────────────────────────────────────────────────────
\node[encblk, right=0.8cm of input]         (enc0) {Enc $s{=}2$\\[-3pt]\tiny $c_0$};
\node[encblk, above=0.28cm of enc0]          (enc1) {Enc $s{=}4$\\[-3pt]\tiny $c_1$};
\node[encblk, above=0.28cm of enc1]          (enc2) {Enc $s{=}8$\\[-3pt]\tiny $c_2$};
\node[encblk, above=0.28cm of enc2]          (enc3) {Enc $s{=}16$\\[-3pt]\tiny $c_3$};

\begin{scope}[on background layer]
  \node[grpbox=encblue, fit=(enc0)(enc3),
        label={[anchor=south, color=encblue, font=\scriptsize\bfseries]north:%
               Encoder (Swin-T \textbar{} ResNet50-3D)}]
        (encbox) {};
\end{scope}

\node[below=0.18cm of enc0, lbl, align=center]
  {Swin: 48/96/192/384\\ResNet50: 256/512/1024/2048};

% ── FPN ──────────────────────────────────────────────────────────────────
\node[fpnblk, right=1.0cm of enc3] (fpn3) {Lat$_3$\\[-3pt]\tiny $p_3$};
\node[fpnblk, right=1.0cm of enc2] (fpn2) {Lat$_2{\oplus}{\uparrow}$\\[-3pt]\tiny $p_2$};
\node[fpnblk, right=1.0cm of enc1] (fpn1) {Lat$_1{\oplus}{\uparrow}$\\[-3pt]\tiny $p_1$};
\node[fpnblk, right=1.0cm of enc0] (fpn0) {Lat$_0{\oplus}{\uparrow}$\\[-3pt]\tiny $p_0$};

\begin{scope}[on background layer]
  \node[grpbox=fpngreen, fit=(fpn0)(fpn3),
        label={[anchor=south, color=fpngreen!60!black, font=\scriptsize\bfseries]north:FPN (top-down)}]
        (fpnbox) {};
\end{scope}

% ── SE BLOCK ─────────────────────────────────────────────────────────────
\node[seblk, right=0.8cm of fpn0] (se)
     {SE Block\\[-3pt]\tiny chan.\ recalib.};

% ── PREDICTION HEADS ─────────────────────────────────────────────────────
\node[wtblk, right=0.8cm of se]  (wt) {WT Head\\[-3pt]\tiny Conv${\times}2 \to 1$ch};
\node[tcblk, below=0.42cm of wt] (tc) {TC Head\\[-3pt]\tiny Conv${\times}2 \to 1$ch};
\node[etblk, below=0.42cm of tc] (et) {ET Head\\[-3pt]\tiny Conv${\times}2 \to 1$ch};

\begin{scope}[on background layer]
  \node[grpbox=headet!80, fit=(wt)(et),
        label={[anchor=south, color=headet!80!black, font=\scriptsize\bfseries]north:HCD Heads}]
        (headbox) {};
\end{scope}

\node[below=0.12cm of et, lbl] {ET $\subseteq$ TC $\subseteq$ WT (soft conditioning)};

% ── OUTPUTS ──────────────────────────────────────────────────────────────
\node[outblk, right=0.8cm of wt, fill=headwt!12, draw=headwt]  (owt) {\textbf{WT}};
\node[outblk, right=0.8cm of tc, fill=headtc!12, draw=headtc]  (otc) {\textbf{TC}};
\node[outblk, right=0.8cm of et, fill=headet!12, draw=headet]  (oet) {\textbf{ET}};

\begin{scope}[on background layer]
  \node[grpbox=darkgray, fit=(owt)(oet),
        label={[anchor=south, color=darkgray, font=\scriptsize\bfseries]north:Logits}]
        (outbox) {};
\end{scope}

% ── FINAL UPSAMPLE ───────────────────────────────────────────────────────
\node[iobox, right=0.7cm of outbox, minimum height=2.0cm, minimum width=1.7cm, align=center]
     (seg) {$\uparrow$ tri-linear\\[-3pt]\tiny $3{\times}H{\times}W{\times}D$};

% ═══════════════════════════════════════════════════════════════════════════
% CONNECTIONS
% ═══════════════════════════════════════════════════════════════════════════

% Input → Encoder
\draw[arr] (input.east) -- (enc0.west);
% (enc0–enc3 built by encoder internally; main feed from input to enc0)

% Encoder → FPN laterals
\draw[arr] (enc0.east) -- (fpn0.west);
\draw[arr] (enc1.east) -- (fpn1.west);
\draw[arr] (enc2.east) -- (fpn2.west);
\draw[arr] (enc3.east) -- (fpn3.west);

% FPN top-down (vertical)
\draw[fpnarr] (fpn3.south) -- (fpn2.north);
\draw[fpnarr] (fpn2.south) -- (fpn1.north);
\draw[fpnarr] (fpn1.south) -- (fpn0.north);

% FPN p0 → SE
\draw[arr] (fpn0.east) -- (se.west);

% SE → WT head (main path)
\draw[arr] (se.east) -- (wt.west);

% SE feature also reaches TC and ET heads
% Route via a right-angled detour to avoid overlap
\coordinate (seJunc) at ($(se.east)!0.4!(wt.west)$);
\draw[arr] (seJunc) |- (tc.west);
\draw[arr] (seJunc) |- (et.west);

% WT → TC conditioning (dashed orange)
\draw[cond] (wt.south) -- node[right=1pt, lbl, color=seorange]
    {$\sigma(l_{WT})$} (tc.north);

% TC → ET conditioning
\draw[cond] (tc.south) -- node[right=1pt, lbl, color=seorange]
    {$\sigma(l_{TC})$} (et.north);

% Heads → logit outputs
\draw[arr] (wt.east) -- (owt.west);
\draw[arr] (tc.east) -- (otc.west);
\draw[arr] (et.east) -- (oet.west);

% Logit box → upsample
\draw[arr] (outbox.east) -- (seg.west);

\end{tikzpicture}
