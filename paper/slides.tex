% ── 2. Problem ────────────────────────────────────────────────────────────────
\begin{frame}{Brain Tumor Segmentation}
  \vspace{0.5em}
  \begin{columns}
    \column{0.5\textwidth}
      \begin{itemize}
        \setlength\itemsep{0.8em}
        \item High-grade gliomas: median survival $\leq$ 2 years
        \item Treatment requires delineating \textbf{three sub-regions} from MRI
        \item Manual segmentation: slow and inconsistent
      \end{itemize}

    \column{0.45\textwidth}
      \begin{block}{Three Sub-Regions}
        \vspace{0.3em}
        \textbf{WT} — Whole Tumor\\[0.4em]
        \textbf{TC} — Tumor Core\\[0.4em]
        \textbf{ET} — Enhancing Tumor
        \vspace{0.3em}
      \end{block}
  \end{columns}
\end{frame}

% ── 3. The Constraint ─────────────────────────────────────────────────────────
\begin{frame}{An Anatomical Fact}
  \vspace{1em}
  \begin{center}
    \Huge $\text{ET} \subset \text{TC} \subset \text{WT}$
  \end{center}

  \vspace{1em}
  \begin{center}
    \large By anatomical definition — in \textbf{every single} BraTS case.
  \end{center}

  \vspace{1.2em}
  \begin{center}
    \textcolor{seorange}{\large
    Yet standard models predict ET, TC, WT \textbf{independently}\\[0.3em]
    and can output ET outside TC.}
  \end{center}
\end{frame}

% ── 4. Prior: Cascade ─────────────────────────────────────────────────────────
\begin{frame}{Prior Work: Cascaded Networks}
  \vspace{0.5em}
  \textbf{Idea:} separate network per sub-region — predict WT, then TC inside WT, then ET inside TC.

  \vspace{1em}
  \begin{alertblock}{Problems}
    \begin{itemize}
      \setlength\itemsep{0.5em}
      \item Each stage trained \textbf{separately} — not end-to-end
      \item Hard binary mask at stage boundary breaks gradient flow
      \item $3\times$ inference cost
    \end{itemize}
  \end{alertblock}
\end{frame}

% ── 5. Prior: Loss-Only ───────────────────────────────────────────────────────
\begin{frame}{Prior Work: Constraint Loss}
  \vspace{0.5em}
  \textbf{Idea:} add a penalty term to the loss that punishes constraint violations during training.

  \vspace{1em}
  \begin{alertblock}{Problem}
    \begin{itemize}
      \setlength\itemsep{0.5em}
      \item The \textbf{architecture itself remains unconstrained}
      \item At inference, with no gradient signal, the model can still freely violate ET $\subset$ TC $\subset$ WT
    \end{itemize}
  \end{alertblock}

  \vspace{1em}
  \begin{center}
    \textbf{Neither approach gives a single network, end-to-end,\\
    with the constraint built into the architecture.}
  \end{center}
\end{frame}

% ── 6. Our Solution ───────────────────────────────────────────────────────────
\begin{frame}{Our Solution: HCD}
  \vspace{0.5em}
  \begin{block}{Hierarchical Consistency Decoder}
    A \textbf{single end-to-end decoder} that enforces ET $\subset$ TC $\subset$ WT
    through \textbf{sequential soft architectural conditioning}.
  \end{block}

  \vspace{1em}
  \begin{center}
  \small
  \begin{tabular}{lccc}
    \toprule
    & \textbf{End-to-end} & \textbf{Constraint in architecture} & \textbf{Single network} \\
    \midrule
    Cascade       & \textcolor{red}{$\times$}                & \textcolor{red}{$\times$}                & \textcolor{red}{$\times$} \\
    Loss-only     & \textcolor{green!50!black}{$\checkmark$} & \textcolor{red}{$\times$}                & \textcolor{green!50!black}{$\checkmark$} \\
    \textbf{HCD (ours)} & \textcolor{green!50!black}{$\checkmark$} & \textcolor{green!50!black}{$\checkmark$} & \textcolor{green!50!black}{$\checkmark$} \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{frame}

% ── 7. Architecture ───────────────────────────────────────────────────────────
\begin{frame}{Architecture}
  \vspace{0.5em}
  \begin{center}
  \large
  \textbf{MRI input}
  $\;\longrightarrow\;$
  \textcolor{encblue}{\textbf{Encoder}}
  $\;\longrightarrow\;$
  \textcolor{fpngreen}{\textbf{FPN}}
  $\;\longrightarrow\;$
  \textcolor{seorange}{\textbf{SE}}
  $\;\longrightarrow\;$
  \textbf{HCD Heads}
  \end{center}

  \vspace{1.2em}
  \begin{columns}
    \column{0.24\textwidth}
      \begin{block}{\textcolor{encblue}{Encoder}}
        \small Extracts features at 4 scales from the MRI volume
      \end{block}
    \column{0.24\textwidth}
      \begin{block}{\textcolor{fpngreen}{FPN}}
        \small Fuses all 4 scales into one rich feature map
      \end{block}
    \column{0.24\textwidth}
      \begin{block}{\textcolor{seorange}{SE Block}}
        \small Recalibrates which feature channels matter
      \end{block}
    \column{0.24\textwidth}
      \begin{block}{HCD Heads}
        \small WT $\to$ TC $\to$ ET, each conditioned on the previous
      \end{block}
  \end{columns}

  \vspace{1em}
  \begin{center}
    Two encoders tested: \textbf{Swin Transformer} and \textbf{ResNet50-3D}
  \end{center}
\end{frame}

% ── 8. Sequential Conditioning ────────────────────────────────────────────────
\begin{frame}{Sequential Soft Conditioning}
  \vspace{0.5em}
  \begin{columns}
    \column{0.48\textwidth}
      \textbf{Standard — independent heads:}
      \vspace{0.3em}
      \begin{align*}
        \hat{y}_{WT} &= \sigma(\text{Head}_{WT}(\tilde{p}_0)) \\
        \hat{y}_{TC} &= \sigma(\text{Head}_{TC}(\tilde{p}_0)) \\
        \hat{y}_{ET} &= \sigma(\text{Head}_{ET}(\tilde{p}_0))
      \end{align*}
      \small No communication between heads.

    \column{0.48\textwidth}
      \textbf{HCD — conditioned heads:}
      \vspace{0.3em}
      \begin{align*}
        l_{WT} &= \text{Head}_{WT}(\tilde{p}_0) \\
        l_{TC} &= \text{Head}_{TC}([\tilde{p}_0,\; \sigma(l_{WT})]) \\
        l_{ET} &= \text{Head}_{ET}([\tilde{p}_0,\; \sigma(l_{TC})])
      \end{align*}
      \small TC sees where WT is. ET sees where TC is.
  \end{columns}

  \vspace{1em}
  \begin{center}
    \textcolor{primary}{Soft masks ($\sigma$ output, 0$\to$1) keep the decoder \textbf{fully differentiable}.}
  \end{center}
\end{frame}

% ── 12. Conclusion ────────────────────────────────────────────────────────────
\begin{frame}{Conclusion}
  \vspace{0.5em}
  \begin{block}{HCD}
    The first single end-to-end decoder that enforces ET $\subset$ TC $\subset$ WT
    \textbf{architecturally} — not post-processing, not loss alone.
  \end{block}

  \vspace{1.2em}
  \begin{itemize}
    \setlength\itemsep{0.8em}
    \item \textbf{vs.\ Cascade} — single network, end-to-end, same inference cost
    \item \textbf{vs.\ Loss-only} — constraint holds at inference, not just during training
    \item \textbf{Encoder-agnostic} — works with both Swin Transformer and ResNet50-3D
  \end{itemize}
